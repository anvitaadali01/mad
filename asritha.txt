package com.example.autocompleteapp

import android.os.Bundle
import android.widget.ArrayAdapter
import android.widget.AutoCompleteTextView
import android.widget.Button
import android.widget.Toast
import androidx.appcompat.app.AppCompatActivity

class MainActivity : AppCompatActivity() {

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)

        val autoText = findViewById<AutoCompleteTextView>(R.id.autoText)
        val btnCheck = findViewById<Button>(R.id.btnCheck)

        val languages = arrayOf(
            "Java",
            "Kotlin",
            "Python",
            "C",
            "C++",
            "JavaScript"
        )

        val adapter = ArrayAdapter(
            this,
            android.R.layout.simple_dropdown_item_1line,
            languages
        )

        autoText.setAdapter(adapter)

        btnCheck.setOnClickListener {

            val enteredText = autoText.text.toString()

            if (languages.contains(enteredText)) {
                Toast.makeText(
                    this,
                    "Language Matched Successfully",
                    Toast.LENGTH_SHORT
                ).show()
            } else {
                autoText.error = "Invalid Language"
            }
        }
    }
}

package com.example.autocompleteapp

import android.os.Bundle
import android.widget.ArrayAdapter
import android.widget.AutoCompleteTextView
import android.widget.Button
import android.widget.Toast
import androidx.appcompat.app.AppCompatActivity

class MainActivity : AppCompatActivity() {

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)

        val autoText = findViewById<AutoCompleteTextView>(R.id.autoText)
        val btnCheck = findViewById<Button>(R.id.btnCheck)

        val languages = arrayOf(
            "Java",
            "Kotlin",
            "Python",
            "C",
            "C++",
            "JavaScript"
        )

        val adapter = ArrayAdapter(
            this,
            android.R.layout.simple_list_item_1,
            languages
        )

        autoText.setAdapter(adapter)
        autoText.threshold = 1

        btnCheck.setOnClickListener {

            val enteredText = autoText.text.toString()

            if (languages.contains(enteredText)) {
                Toast.makeText(
                    this,
                    "Language Matched Successfully",
                    Toast.LENGTH_SHORT
                ).show()
            } else {
                autoText.error = "Invalid Language"
            }
        }
    }
}
