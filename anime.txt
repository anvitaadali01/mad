# main KT
package com.example.animationapp

import android.os.Bundle
import android.view.animation.Animation
import android.view.animation.AnimationUtils
import android.widget.Button
import android.widget.TextView
import androidx.appcompat.app.AppCompatActivity

class MainActivity : AppCompatActivity() {

    lateinit var text: TextView

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)

        text = findViewById(R.id.textView)

        val fadeIn = AnimationUtils.loadAnimation(this, R.anim.fade_in)
        val fadeOut = AnimationUtils.loadAnimation(this, R.anim.fade_out)
        val blink = AnimationUtils.loadAnimation(this, R.anim.blink)
        val zoomIn = AnimationUtils.loadAnimation(this, R.anim.zoom_in)
        val zoomOut = AnimationUtils.loadAnimation(this, R.anim.zoom_out)
        val rotate = AnimationUtils.loadAnimation(this, R.anim.rotate)
        val slideUp = AnimationUtils.loadAnimation(this, R.anim.slide_up)
        val slideDown = AnimationUtils.loadAnimation(this, R.anim.slide_down)

        findViewById<Button>(R.id.fade_in).setOnClickListener {
            text.startAnimation(fadeIn)
        }

        findViewById<Button>(R.id.fade_out).setOnClickListener {
            text.startAnimation(fadeOut)
        }

        findViewById<Button>(R.id.zoom_in).setOnClickListener {
            text.startAnimation(zoomIn)
        }

        findViewById<Button>(R.id.zoom_out).setOnClickListener {
            text.startAnimation(zoomOut)
        }

        findViewById<Button>(R.id.slide_up).setOnClickListener {
            text.startAnimation(slideUp)
        }

        findViewById<Button>(R.id.slide_down).setOnClickListener {
            text.startAnimation(slideDown)
        }

        findViewById<Button>(R.id.bounce).setOnClickListener {
            text.startAnimation(blink)   // using blink as bounce alternative
        }

        findViewById<Button>(R.id.rotate).setOnClickListener {
            text.startAnimation(rotate)
        }

        // 🔥 CROSSFADE (special case)
        text.setOnClickListener {
            text.animate().alpha(0f).setDuration(500).withEndAction {
                text.animate().alpha(1f).duration = 500
            }
        }
    }
}
